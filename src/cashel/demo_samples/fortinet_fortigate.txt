config firewall policy
    edit 1
        set name "Internet-to-DMZ-Web"
        set srcintf "wan1"
        set dstintf "dmz"
        set srcaddr "all"
        set dstaddr "GRP-WEB-SERVERS"
        set action accept
        set schedule "always"
        set service "HTTP" "HTTPS"
        set utm-status enable
        set ssl-ssh-profile "certificate-inspection"
        set av-profile "default"
        set ips-sensor "default"
        set webfilter-profile "default"
        set logtraffic all
        set logtraffic-start enable
        set comments "Allow inbound HTTP/HTTPS to web tier"
    next
    edit 2
        set name "DMZ-to-App"
        set srcintf "dmz"
        set dstintf "lan"
        set srcaddr "GRP-WEB-SERVERS"
        set dstaddr "APP-SERVERS"
        set action accept
        set schedule "always"
        set service "SVC-APP-HTTP"
        set logtraffic all
        set comments "Web tier to application tier"
    next
    edit 3
        set name "App-to-DB"
        set srcintf "lan"
        set dstintf "lan"
        set srcaddr "APP-SERVERS"
        set dstaddr "DB-SERVERS"
        set action accept
        set schedule "always"
        set service "MS-SQL" "POSTGRESQL"
        set logtraffic all
        set comments "Application tier to database tier"
    next
    edit 4
        set name "Inside-to-Internet"
        set srcintf "lan"
        set dstintf "wan1"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "HTTP" "HTTPS" "DNS" "SMTP" "SMTPS"
        set utm-status enable
        set ssl-ssh-profile "certificate-inspection"
        set av-profile "default"
        set ips-sensor "default"
        set webfilter-profile "default"
        set logtraffic all
        set comments "Permitted egress with UTM inspection"
    next
    edit 5
        set name "Allow-All-Unrestricted"
        set srcintf "any"
        set dstintf "any"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "ALL"
        set logtraffic disable
        set comments "AUDIT FINDING: any/any with ALL services and logging disabled"
    next
    edit 6
        set name "Allow-Web-Duplicate"
        set srcintf "wan1"
        set dstintf "dmz"
        set srcaddr "all"
        set dstaddr "GRP-WEB-SERVERS"
        set action accept
        set schedule "always"
        set service "HTTP" "HTTPS"
        set utm-status enable
        set ssl-ssh-profile "certificate-inspection"
        set av-profile "default"
        set ips-sensor "default"
        set logtraffic all
        set comments "AUDIT FINDING: duplicate of policy 1 — shadowed/redundant"
    next
    edit 7
        set name "Mgmt-Access"
        set srcintf "lan"
        set dstintf "wan1"
        set srcaddr "MGMT-HOSTS"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "SSH" "HTTPS"
        set logtraffic all
        set comments "Management host outbound access"
    next
    edit 8
        set name "Implicit-Deny"
        set srcintf "any"
        set dstintf "any"
        set srcaddr "all"
        set dstaddr "all"
        set action deny
        set schedule "always"
        set service "ALL"
        set logtraffic all
        set comments "Explicit deny-all with logging"
    next
end
