! --- Access Control Lists ---
!
! OUTSIDE_IN: Traffic entering from the internet to DMZ / inside
access-list OUTSIDE_IN extended permit tcp any object-group OBJ_WEB_SERVERS object-group OBJ_SVC_WEB log
access-list OUTSIDE_IN extended permit icmp any object-group OBJ_WEB_SERVERS echo-reply log
access-list OUTSIDE_IN extended permit icmp any object-group OBJ_WEB_SERVERS unreachable log
access-list OUTSIDE_IN extended permit icmp any object-group OBJ_WEB_SERVERS time-exceeded log
! NOTE: Overly permissive — any/any permit — AUDIT WILL FLAG THIS
access-list OUTSIDE_IN extended permit ip any any log
! NOTE: Missing log keyword on this rule — AUDIT WILL FLAG THIS
access-list OUTSIDE_IN extended permit tcp any host 172.16.10.20 eq 8080
access-list OUTSIDE_IN extended deny ip any any log
!
! INSIDE_OUT: Traffic from inside to internet
access-list INSIDE_OUT extended permit tcp 10.10.0.0 255.255.0.0 any object-group OBJ_SVC_WEB log
access-list INSIDE_OUT extended permit udp 10.10.0.0 255.255.0.0 any eq 53 log
access-list INSIDE_OUT extended permit tcp 10.10.0.0 255.255.0.0 any eq 25 log
access-list INSIDE_OUT extended permit tcp 10.10.0.0 255.255.0.0 any eq 587 log
! NOTE: Unrestricted ICMP to any — AUDIT WILL FLAG THIS
access-list INSIDE_OUT extended permit icmp any any log
access-list INSIDE_OUT extended deny ip any any log
!
! DMZ_IN: Traffic from DMZ to inside app/db tiers
access-list DMZ_IN extended permit tcp object-group OBJ_WEB_SERVERS object-group OBJ_APP_SERVERS eq 8080 log
access-list DMZ_IN extended permit tcp object-group OBJ_APP_SERVERS object-group OBJ_DB_SERVERS object-group OBJ_SVC_DB log
! NOTE: Redundant duplicate rule — AUDIT WILL FLAG THIS
access-list DMZ_IN extended permit tcp object-group OBJ_WEB_SERVERS object-group OBJ_APP_SERVERS eq 8080 log
access-list DMZ_IN extended deny ip any any log
!
! MGMT_IN: Management plane access control
access-list MGMT_IN extended permit tcp object-group OBJ_MGMT_HOSTS host 10.10.0.1 object-group OBJ_SVC_MGMT log
access-list MGMT_IN extended deny ip any any log
